module tb_avanco;

reg avanca;
reg clock;
reg reset;
reg [2:0] orientacao;
wire [2:0] acao;

// Instancia o módulo que será testado
avanco uut (
    .avanca(avanca),
    .clock(clock),
    .reset(reset),
    .orientacao(orientacao),
    .acao(acao)
);

// Gera o clock
initial begin
    clock = 0;
    forever #5 clock = ~clock; // Gera um clock com período de 10 unidades de tempo
end

// Testa o módulo
initial begin
    // Inicializa os sinais
    reset = 1;
    avanca = 0;
    orientacao = 3'b000;

    // Espera alguns ciclos de clock
    #10 reset = 0; // Libera o reset

    // Testa a orientação Norte
    #10 orientacao = 3'b001;
    #10 avanca = 1;
    #10 avanca = 0;

    // Testa a orientação Oeste
    #10 orientacao = 3'b010;
    #10 avanca = 1;
    #10 avanca = 0;

    // Testa a orientação Leste
    #10 orientacao = 3'b011;
    #10 avanca = 1;
    #10 avanca = 0;

    // Testa a orientação Sul
    #10 orientacao = 3'b100;
    #10 avanca = 1;
    #10 avanca = 0;

    // Testa o reset durante o avanço
    #10 reset = 1;
    #10 reset = 0;

    // Termina a simulação
    #20 $finish;
end

// Monitor para observar os sinais
initial begin
    $monitor("Time = %0t, Reset = %b, Avanca = %b, Orientacao = %b, Acao = %b", $time, reset, avanca, orientacao, acao);
end

endmodule
